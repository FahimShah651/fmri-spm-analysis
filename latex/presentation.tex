\section{Introduction}
% ==========================================================================

\begin{frame}{Why fMRI?}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{itemize}
        \item Brain activity $\Rightarrow$ increased oxygen demand $\Rightarrow$ more blood flow
        \item Oxyhaemoglobin vs.\ deoxyhaemoglobin have different MRI signals
        \item This difference is the \textbf{BOLD response} ($\sim$1--5\% signal change)
        \item Non-invasive, whole-brain, \textasciitilde 2\,s temporal resolution
        \item Same scanner session can probe completely different brain systems
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{BOLD signal chain}
        Stimulus\\
        $\downarrow$ Neural firing\\
        $\downarrow$ Local blood flow $\uparrow$\\
        $\downarrow$ Oxyhaemoglobin $\uparrow$\\
        $\rightarrow$ MRI signal change\\[0.3em]
        \small Ogawa et al., 1990
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Project Goal and Approach}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{itemize}
        \item Build a \textbf{complete fMRI pipeline} from raw NIfTI $\to$ connectivity maps
        \item Implemented in \textbf{MATLAB + SPM25} with scripted batch processing
        \item Two independent datasets to cross-validate the pipeline
        \item Skip-if-exists logic: each step re-runnable independently
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{Seven-step pipeline}
        \texttt{step01} --- Environment \& download\\
        \texttt{step02} --- DS1 preprocessing\\
        \texttt{step03} --- DS2 preprocessing\\
        \texttt{step04} --- DS1 GLM\\
        \texttt{step05} --- DS2 GLM\\
        \texttt{step06} --- Connectivity\\
        \texttt{step07} --- Comparison \& QC
      \end{block}
  \end{columns}
\end{frame}

\section{Preprocessing Pipeline}
% ==========================================================================

\begin{frame}{Preprocessing: Six-Step Pipeline}
  \begin{columns}
    \column{0.5\textwidth}
      \textbf{Why preprocessing?}
      \begin{itemize}
        \item Slices acquired at different times within TR
        \item Head motion mimics task-related BOLD signal
        \item Need common space (MNI152) for interpretation
      \end{itemize}
      \vspace{0.5em}
      \small SPM file prefix chain:\\
      \texttt{bold} $\to$ \texttt{a} $\to$ \texttt{ra} $\to$ \texttt{wra} $\to$ \texttt{swra}
    \column{0.5\textwidth}
      \begin{block}{Same pipeline applied to both datasets}
        \textbf{1.} Slice Timing Correction\\
        \textbf{2.} Realignment (motion correction)\\
        \textbf{3.} Coregistration (EPI $\leftrightarrow$ T1)\\
        \textbf{4.} Unified Segmentation (tissue + deformation)\\
        \textbf{5.} Normalisation to MNI152 (2\,mm)\\
        \textbf{6.} Spatial Smoothing (Gaussian 6\,mm FWHM)
      \end{block}
  \end{columns}
\end{frame}

\section{General Linear Model}
% ==========================================================================

\begin{frame}{The General Linear Model}
  \begin{columns}
    \column{0.55\textwidth}
      \[
        \mathbf{Y} = \mathbf{X}\boldsymbol{\beta} +
        \boldsymbol{\varepsilon}, \quad
        \boldsymbol{\varepsilon} \sim \mathcal{N}(0, \sigma^2\mathbf{V})
      \]
      \begin{itemize}
        \item $\mathbf{Y}$: BOLD time series at each voxel
        \item $\mathbf{X}$: design matrix (task + nuisance regressors)
        \item $\boldsymbol{\beta}$: how strongly each condition drives a voxel
        \item t-contrast: tests if specific $\boldsymbol{\beta} > 0$
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{SPM25 GLM settings}
        \begin{itemize}
          \item Onsets from BIDS \texttt{events.tsv}
          \item Canonical double-gamma HRF
          \item 6 motion parameters (nuisance)
          \item 128\,s high-pass filter (DCT)
          \item AR(1) temporal autocorrelation
          \item Threshold: $p<0.001$ uncorr., $k>10$\,vox
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\section{Functional Connectivity}
% ==========================================================================

\begin{frame}{Seed-Based Functional Connectivity}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{Key steps:}
      \begin{enumerate}
        \item Spherical seed ROI (6\,mm radius)
        \item Extract mean seed time series
        \item High-pass filter ($>0.01$\,Hz)
        \item Pearson correlation $r$ with every brain voxel
        \item Fisher $z$-transform: $z(r) = \tanh^{-1}(r)$
        \item Threshold at $r > 0.30$ for display
      \end{enumerate}
    \column{0.45\textwidth}
      \begin{block}{Seed coordinates}
        \textbf{DS1 --- Left M1}\\
        MNI $[-38, -26, 60]$\,mm\\
        (hand area)\\[0.5em]
        \textbf{DS2 --- Primary Visual Cortex}\\
        MNI $[0, -88, 2]$\,mm\\
        (calcarine sulcus, V1)
      \end{block}
      \vspace{0.3em}
      \small Note: task run used (not resting-state)
  \end{columns}
\end{frame}
