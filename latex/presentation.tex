\section{Introduction}
% ==========================================================================

\begin{frame}{Why fMRI?}
  \begin{columns}
    \column{0.6\textwidth}
      \textbf{The core idea is beautifully simple.}\\[0.5em]
      When a group of neurons fires, they need more oxygen. Blood flow
      increases to meet that demand, and the ratio of oxygenated to
      deoxygenated haemoglobin changes. Because oxygenated haemoglobin is
      slightly diamagnetic, the MRI scanner can detect this difference.
      That tiny change in signal is the \textbf{BOLD response}.
      \\[0.8em]
      What makes fMRI special is that we can ask the same brain, in the
      same scanner session, about completely different cognitive systems
      just by changing the task.
    \column{0.4\textwidth}
      \begin{block}{BOLD signal model}
        \vspace{0.3em}
        Stimulus $\rightarrow$ Neural activity\\[0.2em]
        $\downarrow$ Neural activity\\[0.2em]
        $\uparrow$ Local blood flow\\[0.2em]
        $\uparrow$ Oxyhaemoglobin\\[0.2em]
        $\rightarrow$ MRI signal change ($\sim$1--5\%)
        \vspace{0.3em}
      \end{block}
      \vspace{0.5em}
      \small First described by Ogawa et al., 1990.
  \end{columns}
\end{frame}

\begin{frame}{Project Goal and Approach}
  \begin{columns}
    \column{0.55\textwidth}
      The goal was to build a \textbf{complete, scripted fMRI pipeline}
      from scratch using MATLAB and SPM25 — not just run a GUI, but
      understand every step from the raw NIfTI files through to
      connectivity maps.
      \\[0.8em]
      The pipeline is split into \textbf{seven numbered MATLAB scripts}
      that run in sequence. Each step saves its output and skips
      re-processing if the output already exists, so individual steps
      can be re-run or debugged independently.
      \\[0.8em]
      Two datasets were chosen deliberately to cover different functional
      systems: one motor, one visual. Running both through the same code
      gives us a natural sanity check.
    \column{0.45\textwidth}
      \begin{block}{Seven-step pipeline}
        \texttt{step01} — Environment \& download\\
        \texttt{step02} — DS1 preprocessing\\
        \texttt{step03} — DS2 preprocessing\\
        \texttt{step04} — DS1 GLM\\
        \texttt{step05} — DS2 GLM\\
        \texttt{step06} — Connectivity\\
        \texttt{step07} — Comparison \& QC
      \end{block}
  \end{columns}
\end{frame}

\section{Preprocessing Pipeline}
% ==========================================================================

\begin{frame}{Preprocessing: Why It Matters}
  \begin{columns}
    \column{0.5\textwidth}
      Raw fMRI data straight off the scanner has three fundamental
      problems:
      \\[0.6em]
      \textbf{1. Time problem:} Each brain slice is acquired at a
      slightly different moment within the TR window. The last slice
      lags the first by almost a full TR.
      \\[0.4em]
      \textbf{2. Motion problem:} Subjects move. Even 0.5\,mm of
      head movement correlated with the task can produce fake
      activations that look exactly like real BOLD responses.
      \\[0.4em]
      \textbf{3. Space problem:} Functional and anatomical images
      don't line up. We need everything in a common standard space
      (MNI152) before we can compare results across analyses.

    \column{0.5\textwidth}
      \begin{block}{Six preprocessing steps (same for both datasets)}
        \textbf{1.} Slice Timing Correction\\
        \textbf{2.} Realignment (motion correction)\\
        \textbf{3.} Coregistration (EPI $\leftrightarrow$ T1)\\
        \textbf{4.} Unified Segmentation (tissue + deformation)\\
        \textbf{5.} Normalisation to MNI152 (2\,mm)\\
        \textbf{6.} Spatial Smoothing (Gaussian 6\,mm FWHM)
      \end{block}
      \vspace{0.3em}
      \small SPM prefix convention after each step:\\
      \texttt{bold} $\to$ \texttt{a} $\to$ \texttt{ra} $\to$
      \texttt{wra} $\to$ \texttt{swra}
  \end{columns}
\end{frame}

\section{General Linear Model}
% ==========================================================================

\begin{frame}{The General Linear Model}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{The central idea:} model the BOLD time series at every
      voxel as a linear combination of task regressors plus noise.
      \[
        \mathbf{Y} = \mathbf{X}\boldsymbol{\beta} +
        \boldsymbol{\varepsilon}, \quad
        \boldsymbol{\varepsilon} \sim \mathcal{N}(0, \sigma^2\mathbf{V})
      \]
      \begin{itemize}
        \item $\mathbf{Y}$: time series of a single voxel
        \item $\mathbf{X}$: design matrix (task + nuisance regressors)
        \item $\boldsymbol{\beta}$: parameter estimates (``how much
              does each condition drive this voxel?'')
        \item $\boldsymbol{\varepsilon}$: noise (modelled with AR(1))
      \end{itemize}
    \column{0.45\textwidth}
      \begin{block}{SPM25 GLM specification}
        \begin{itemize}
          \item Onsets from BIDS \texttt{events.tsv}
          \item Canonical double-gamma HRF convolution
          \item 6 motion parameters as nuisance columns
          \item 128\,s high-pass filter (DCT basis)
          \item AR(1) temporal autocorrelation
        \end{itemize}
      \end{block}
      \vspace{0.5em}
      \small A \textbf{t-contrast} tests whether a specific
      $\boldsymbol{\beta}$ (or linear combination) is significantly
      positive, producing a t-statistic map across the brain.
  \end{columns}
\end{frame}

\section{Functional Connectivity}
% ==========================================================================

\begin{frame}{Seed-Based Functional Connectivity — Method}
  \begin{columns}
    \column{0.55\textwidth}
      Seed-based connectivity asks a simple question: \textit{which
      other brain regions fluctuate in sync with my chosen seed?}
      \\[0.5em]
      \textbf{Procedure:}
      \begin{enumerate}
        \item Define a spherical seed ROI (radius = 6\,mm)
        \item Extract the mean time series across all seed voxels
        \item High-pass filter the seed signal ($> 0.01$\,Hz)
        \item Compute Pearson correlation $r$ between the seed
              and every other brain voxel
        \item Apply Fisher $z$-transform to stabilise variance:
        \[
          z(r) = \tanh^{-1}(r) =
          \tfrac{1}{2}\ln\!\left(\tfrac{1+r}{1-r}\right)
        \]
        \item Threshold at $r > 0.30$ for display
      \end{enumerate}
    \column{0.45\textwidth}
      \begin{block}{Seed coordinates}
        \textbf{DS1 — Left M1}\\
        MNI $[-38, -26, 60]$\,mm\\
        (Hand area, well-established)\\[0.6em]
        \textbf{DS2 — Primary Visual Cortex}\\
        MNI $[0, -88, 2]$\,mm\\
        (Calcarine sulcus, V1 centre)
      \end{block}
      \vspace{0.4em}
      \small Note: this analysis uses the full task run, not a
      resting-state scan. The correlation structure partially reflects
      task-related coactivation.

\begin{frame}{Dataset 1 — Motor Network Connectivity}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{figure}
        \includegraphics[width=\textwidth]{fig21_ds1_connectivity_M1_hand}
        \caption{Seed-based connectivity from left M1
          ($[-38, -26, 60]$\,mm). Warm = positive correlation
          ($r>0.30$). Cool = anticorrelated regions.}
      \end{figure}
    \column{0.45\textwidth}
      Regions positively correlated with the left M1 seed form a
      recognisable \textbf{sensorimotor network}:
      \\[0.4em]
      \begin{itemize}
        \item Bilateral primary motor cortex
        \item Supplementary motor area (SMA) on the medial wall
        \item Premotor cortex (anterior to M1)
        \item Thalamus (basal ganglia-M1 relay)
      \end{itemize}
      \vspace{0.4em}
      This matches the classic motor resting-state network first
      reported by Biswal et al.\ (1995).
      \\[0.4em]
      The \textbf{cool regions} are the default mode network
      (medial prefrontal, posterior cingulate) — an expected
      anti-correlation, since motor tasks suppress default-mode
      activity.
  \end{columns}
\end{frame}

      \begin{itemize}
        \item The fusiform gyrus (ventral stream)
        \item The lingual gyrus
        \item Dorsal parietal cortex (dorsal stream)

      co-activate every time a visual stimulus appears on screen.

      Negative correlations are again found in \textbf{medial
      prefrontal and posterior cingulate} — the default mode
      suppresses during active visual processing.
  \end{columns}

\begin{frame}{Cross-Dataset Connectivity Comparison}
  \begin{figure}

    \caption{Side-by-side connectivity: DS1 motor network (top) and
      DS2 visual network (bottom) at a shared axial slice.
      Warm = positive correlation ($r>0.30$). Cool = negative.}
  \end{figure}
  \vspace{0.2em}
  \small \textbf{The spatial separation here is the key result.} The motor seed produces
  connectivity in central and frontal areas (top), while the visual seed is constrained
  to occipital and temporal areas (bottom). These two networks don't overlap — exactly
  what we would expect from two independent functional systems.
\end{frame}

\section{Conclusion}
% ==========================================================================

\begin{frame}{What We Learned and Limitations}
  \begin{columns}
    \column{0.55\textwidth}
      \textbf{The pipeline works, and the results make sense.}
      \\[0.4em]
      \begin{itemize}
        \item Motion was low in both datasets ($<0.2$\,mm mean FD)
        \item Coregistration and normalisation quality was good
        \item Motor somatotopy (finger/foot/lips) matched the
              known homuncular organisation exactly
        \item Face-selective fusiform activation replicated
              Haxby et al.\ (2001)
        \item Motor and visual connectivity networks were spatially
              non-overlapping, as expected
      \end{itemize}
    \column{0.45\textwidth}
      \begin{alertblock}{Limitations}
        \begin{itemize}
          \item Single subject per dataset — can't generalise
          \item No group-level (second-level) model
          \item Connectivity on task data, not resting state
          \item No global signal or WM/CSF regression
          \item Thresholds are uncorrected ($p<0.001$)
        \end{itemize}
      \end{alertblock}
      \vspace{0.4em}
      \begin{block}{Future directions}
        Group analyses, resting-state data, FDR correction,
        dynamic connectivity, MVPA
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Conclusion}
  \begin{columns}
    \column{0.6\textwidth}
      Building this pipeline from scratch was the best way to actually
      understand what each preprocessing step does and why it matters.
      \\[0.6em]
      The key engineering decision — keeping everything sequential,
      scripted, and skip-if-exists — made it practical to debug
      individual steps without re-running the entire pipeline.
      \\[0.6em]
      Validating against two datasets with known expected results
      (motor somatotopy in DS1, fusiform face area in DS2) gives much
      more confidence in the pipeline than testing on a single
      experiment alone.

\end{frame}
